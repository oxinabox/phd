\documentclass{article}
\usepackage{pgfplotstable}


\DeclareMathVersion{tabularbold}
\SetSymbolFont{operators}{tabularbold}{OT1}{cmr}{b}{n}
\begin{document}


\pgfplotstabletypeset[
	precision=5, fixed zerofill=true,
	every row 2 column 1/.style={
		postproc cell content/.append style={
			@cell content/.add={$\bf}{$}
		}
	}
]{
a b
5 1.23
6 1.6311
7 4.231
9 2.10144
}

\vspace{2cm}

\pgfplotstabletypeset[
precision=5, fixed zerofill=true,
boldcell/.style = {
	postproc cell content/.append style={
		@cell content/.add={\mathversion{tabularbold}}{}
	}
},
italiccell/.style = {
	postproc cell content/.append style={
		@cell content/.add={$\it}{$}
	}
},
every row 2 column 1/.style={boldcell},
every row 3 column 1/.style={italiccell}
]{
a b
5 1.23
6 1.6311
7 4.231
9 2.10144
}



\end{document}
