\documentclass[paper=a0paper,landscape,dvipsnames,papersize={48in,36in}]{umbcposter}
\usepackage{times}
\begin{document}
	
	\newcommand{\mytitle}{
		\begin{tabular}{c}
			\LARGE\sc This is the title \\[0.6ex]
			\Large\sl Rouben Rostamian\\
			\Large UMBC
		\end{tabular}
	}
	
	\posterinit{
		%grid,
		background style = {left color = Apricot, right color = white},
		title = {\mytitle},
		right logo,       % use default
		left logo,        % use default
		box/border style, % use default
		box/header style, % use default
		box/body style = {bottom color=blue!10, top color=red!5},
		box/all rounded,
	}
	
	\boxit{col = 0, at top, name=box1}{Title of Box 1}{
		\begin{itemize}
			\item col = 0
			\item at top
			\item name = box1
		\end{itemize}
	}
	
	\boxit{col = 0, below of=box1, name=box2}{Title of Box 2}{
		\begin{itemize}
			\item col = 0
			\item below of = box1
			\item name = box2
		\end{itemize}
	}
	
	\boxit{col = 0, at bottom, name=box3}{Title of Box 3}{
		\begin{itemize}
			\item col = 0
			\item at bottom
			\item name = box3
		\end{itemize}
	}
	
	\boxit{col = 0, below of=box2, above of=box3}{Title of Box 4}{
		\begin{itemize}
			\item col = 0
			\item below of = box2
			\item above of = box3
		\end{itemize}
	}
	
	\boxit{at top, col = 1, span=2, name=box5}{Title of Box 5}{
		\begin{itemize}
			\item at top
			\item col = 1
			\item span = 2
			\item name = box5
		\end{itemize}
	}
	
	\boxit{below of = box5, col=1, name=box6}{Title of Box 6}{
		\begin{itemize}
			\item below of = box5
			\item col = 1
			\item name = box6
		\end{itemize}
	}
	
	\boxit{below of = box6, col=1, span=2.3,
		above of = bottom, name=box7}{Title of Box 7}{
		\begin{itemize}
			\item below of = box6
			\item col = 1
			\item span = 2.3
			\item above of = bottom
			\item name = box7
		\end{itemize}
	}
	
	\boxit{below of = box5, col=2, above of=box7}{Title of Box 8}{
		\begin{itemize}
			\item below of = box5
			\item above of = box7
			\item col = 2
		\end{itemize}
	}
	
	\boxit{at top, col = 3, above of=box7, name=box9}{Title of Box 9}{
		\begin{itemize}
			\item at top
			\item col = 3
			\item above of = box7
			\item name = box9
		\end{itemize}
	}
	
	\boxit{below of = box9, col=3.3, span=0.7, above of=bottom}{Box 10}{
		\begin{itemize}
			\item below of = box9
			\item col = 3.3
			\item span = 0.7
		\end{itemize}
	}
	
\end{document}
